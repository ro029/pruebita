#Funcón decoradora
def convertir_mayus(funcion_a_decorar):
    def func_wrapper(diccionario):
        cad = funcion_a_decorar(diccionario)
        return cad.upper()
    return func_wrapper
        
#Función a decorar
@convertir_mayus
def formatear_xml(diccionario):
    cad = "<persona>\n"
    for llave, valor in diccionario.items():
        cad = cad + "\t<{}>{}</{}>\n".format(llave, valor, llave)
    cad = cad + "</persona>"
    return cad

def main():
    #se pasa la función sin ejecutar--si se deseara ejecutar la función, pusiera "operacion(sumar(), op1, op2)
    persona = {
            "nombre":"Walter",
            "edad":25
    }
    cad_resp = formatear_xml(persona)    
    print(cad_resp)
    
if __name__=="__main__":
    main()
